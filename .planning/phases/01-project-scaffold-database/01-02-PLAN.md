         ```dockerfile
         COPY ServiceCity/ ./ServiceCity/

         COPY ServiceCity.Data/ ./ServiceCity.Data/

         RUN dotnet publish ServiceCity/ServiceCity.csproj -c Release -o /app/publish --no-restore

         # Stage 2: Runtime
         FROM mcr.microsoft.com/dotnet/aspnet:10.0 AS runtime
         WORKDIR /app
         EXPOSE 8080

         USER app

         COPY --from=build /app/publish ./

         ENTRYPOINT ["dotnet", "ServiceCity.dll"]
         ```

ServiceCity.Core/ and ServiceCity.Data/ are sibling directories, but since the build context is `./ServiceCity`, docker cannot access `ServiceCity.Core/` from the root.
